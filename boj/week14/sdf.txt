#include <string>
#include <vector>

using namespace std;

int solution(vector<int> queue1, vector<int> queue2) {
    int answer = 300000+1;
    int n = queue1.size();
    
    vector<long long> v;
    long long sum = 0LL;
    for(int i=0; i<n; i++) {
        v.push_back((long long)queue1[i]);
        sum += v[i];
    }
    for(int i=0; i<n; i++) {
        v.push_back((long long)queue2[i]);
    }
    if(v[n+n-1]%2 == 1) return -1;
    
    for(int i=0; i<v.size(); i++) {
        cout<<v[i]<<' ';
    }
    long long mid = v[n+n-1]/2;
    if(sum == mid) return 0;
    int j = 1;
    for(int i=0; i<n+n+n; i++) {
        sum = 0LL;
        for(j-=1; j<v.size(); j++) {
            sum += v[j];
            if(sum == mid) {
                cout<<'\n'<<i<<' '<<j;
                if(j<n) {
                    cout<<" 1";
                }else if(i<n && j<n+n) {
                    cout<<" 2";
                }
                break;
            }else if(sum > mid) {
                break;
            }
        }
        v.push_back(v[i]);
        
    }
    return answer;
}
